使用Python进行数据清洗、分析和可视化，展示数据分析的完整流程。

python_demo.py 
# 数据分析示例
import pandas as pd
import numpy as np

# 创建示例数据
data = {
    '姓名': ['张三', '李四', '王五', '赵六'],
    '年龄': [25, 30, 35, 40],
    '薪资': [50000, 60000, 70000, 80000],
    '部门': ['技术', '市场', '技术', '财务']
}

df = pd.DataFrame(data)
print("原始数据:")
print(df)
print("\n基本统计:")
print(df.describe())
print("\n按部门分组平均薪资:")
print(df.groupby('部门')['薪资'].mean())

🤖 机器学习示例

简单的线性回归模型示例，展示机器学习的基本流程。

ml_demo.py 
# 线性回归示例
import numpy as np
from sklearn.linear_model import LinearRegression
import matplotlib.pyplot as plt

# 生成示例数据
X = np.array([1, 2, 3, 4, 5]).reshape(-1, 1)
y = np.array([2, 4, 5, 4, 5])

# 创建并训练模型
model = LinearRegression()
model.fit(X, y)

# 预测
y_pred = model.predict(X)

print("模型斜率:", model.coef_[0])
print("模型截距:", model.intercept_)
print("R²分数:", model.score(X, y))

# 可视化结果
plt.scatter(X, y, color='blue')
plt.plot(X, y_pred, color='red')
plt.title('线性回归示例')
plt.xlabel('X')
plt.ylabel('y')
plt.show()
